提供三个最常用的基础函数。
Object.extend: 复制成员
        var src = {a: 1};
        var dest = {};
        Object.extend(dest, src); // 复制 src 的成员到 dest
        console.log(dest);
Object.each: 迭代对象或数组
        Object.each([1, 2, 3], function(item, index){
            console.log(item);
        });
        Object.each({a:1, b: 2}, function(item, key){
            console.log(item);
        });
String.format: 字符串格式化
        String.format("{0}转换", 1); //  "1转换"
        String.format("{1}翻译",0,1); // "1翻译"
        String.format("{a}翻译",{a:"也可以"}); // 也可以翻译
        String.format("{{0}}不转换, {0}转换", 1); //  "{0}不转换1转换"
