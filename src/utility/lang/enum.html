定义枚举

API (源码: utility/lang/enum.js)

API 描述 示例
Enum.getName 根据枚举值获取枚举名 
Enum.getName(WeekDay, 0); // 返回 'sunday'
Enum.hasFlags 枚举标记位判断 
                    var Flags = {
                        red: 1 << 1,
                        yellow: 1 << 2,
                        blue: 1 << 3,
                    };
                    Enum.hasFlags(Flags.red | Flags.yellow, Flags.red); // true
Enum.setFlag 设置指定枚举的标记位 
                    var Flags = {
                        red: 1 << 1,
                        yellow: 1 << 2,
                        blue: 1 << 3,
                    };
                    var flag = 0;
                    Enum.setFlag(flag, Flags.red, true);
